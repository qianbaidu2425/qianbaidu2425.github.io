用递归模拟混沌现象

2023/11/5

背景

代码实现（python）

#这个程序的作用是研究X(n+1)=1-kX(n)^2这个方程实现的混沌现象
import matplotlib.pyplot as plt
x = []
y = []
# 先创建一个空的散点图
plt.scatter(x, y)
# 给散点图添加标题，并给坐标轴命名
plt.title('x-k')
plt.xlabel('k')
plt.ylabel('x')
#设置基础参数
step = 0.01
x = 5
k = 0
#k从0递增到2，步长为0.01
while k<=2:
    #先把k的位数定为2位
    k = round(k,2)
    #先迭代500次，得到方程迭代500次后x的数值
    for i in range(1,501):
        x = 1 - k * x * x
    #接下来再迭代100次，并把每次得到的x，用(k,x)坐标的形式画在散点图中
    #如果是单倍周期，那么x在这100次迭代的过程中会稳定在某一个值，否则会在几个值之间跳动
    for i in range(1,101):
        x = 1 - k * x * x
        plt.scatter(k,x,color = 'black',s = 10)
        print(k)
    k = k + step
#接下来实现鼠标点击读取坐标功能：
def onclick(event):
    if event.xdata is not None and event.ydata is not None:
        print(f"点击坐标：({event.xdata}, {event.ydata})")
cid = plt.gcf().canvas.mpl_connect('button_press_event', onclick)
#把画好的散点图表示出来
plt.show()
